```dart
// test/models/song_test.dart
import 'package:flutter_test/flutter_test.dart';
import 'package:singpromfter_app/models/song.dart';
import 'package:singpromfter_app/models/backing_track.dart';

void main() {
  group('Song 직렬화', () {
    test('toJson/fromJson 라운드트립이 동일성을 유지한다', () {
      final original = Song(
        id: 'test-1',
        title: '테스트 곡',
        lyricsPath: '/path/to/lyrics.txt',
        lyricsText: '가사 내용',
        backingTracks: [
          const BackingTrack(slot: 1, fileName: 'test_mr1.mp3', label: 'MR1'),
          const BackingTrack(slot: 2, fileName: 'test_mr2.mp3', label: 'MR2'),
        ],
        createdAt: DateTime(2026, 7, 4),
        updatedAt: DateTime(2026, 7, 4),
      );

      final json = original.toJson();
      final restored = Song.fromJson(json);

      expect(restored.id, original.id);
      expect(restored.title, original.title);
      expect(restored.lyricsText, original.lyricsText);
      expect(restored.backingTracks.length, 2);
      expect(restored.backingTracks[0].slot, 1);
    });

    test('레거시 mrFileName을 slot 1로 마이그레이션한다', () {
      final json = {
        'id': 'legacy-1',
        'title': '구버전 곡',
        'mrFileName': 'old.mp3',
      };
      final song = Song.fromJson(json);
      expect(song.backingTracks.length, 1);
      expect(song.backingTracks.first.slot, 1);
      expect(song.backingTracks.first.fileName, 'old.mp3');
    });

    test('encodeList/decodeList 라운드트립', () {
      final songs = [/* ... */];
      final encoded = Song.encodeList(songs);
      final decoded = Song.decodeList(encoded);
      expect(decoded.length, songs.length);
    });
  });

  group('PrompterSettings', () {
    test('withSongTrackSlot이 곡별 슬롯을 갱신한다', () {
      const settings = PrompterSettings();
      final updated = settings.withSongTrackSlot('song-1', 2);
      expect(updated.trackSlotForSong('song-1'), 2);
      expect(updated.lastSelectedTrackSlot, 2);
    });

    test('잘못된 JSON에 대해 기본값을 반환한다', () {
      final decoded = PrompterSettings.decode('invalid json');
      expect(decoded.fontSizeLevel, 3);
    });
  });
}
```

### 🔴 S5-E2: CI 구축 (GitHub Actions)

- **우선순위**: P0

#### 작업 지시

**1단계: 메인 워크플로우**
```yaml
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [master, main]
  pull_request:
    branches: [master, main]

jobs:
  analyze:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: subosito/flutter-action@v2
        with:
          channel: stable
      - run: flutter pub get
      - run: flutter analyze

  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: subosito/flutter-action@v2
        with:
          channel: stable
      - run: flutter pub get
      - run: flutter test --coverage
      - name: 커버리지 업로드
        uses: codecov/codecov-action@v4
        if: always()

  build-windows:
    runs-on: windows-latest
    needs: [analyze, test]
    steps:
      - uses: actions/checkout@v4
      - uses: subosito/flutter-action@v2
        with:
          channel: stable
      - run: flutter pub get
      - run: flutter build windows --release
```
